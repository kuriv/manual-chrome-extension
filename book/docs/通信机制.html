popup → service worker

更新 popup.html 文件内容如下。

<!DOCTYPE html>
<html lang="zh-Hans">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Demo</title>
    <link rel="stylesheet" type="text/css" href="popup.css">
</head>
<body>
    <button id="popup-to-service-worker">Send</button>
    <h1 id="service-worker-response"></h1>
    <script type="text/javascript" src="popup.js"></script>
</body>
</html>

更新 popup.js 文件内容如下。

const popup_to_service_worker = document.querySelector('#popup-to-service-worker');
popup_to_service_worker.addEventListener('click', () => {
    chrome.runtime.sendMessage(
    {
        type: 'POPUP_TO_SW',
        content: 'Hello World!'
    },
    response => {
        const service_worker_response = document.querySelector('#service-worker-response');
        chrome.runtime.lastError ? service_worker_response.textContent = chrome.runtime.lastError : service_worker_response.textContent = response.content;
    }
  );
});

更新 service-worker.js 文件内容如下。

chrome.runtime.onMessage.addListener((message, sender, sendResponse) => {
    if (message.type == 'POPUP_TO_SW') {
        console.log(message.content);
        sendResponse({
            content: message.content
        });
    }
});

更新 popup.js 文件内容如下。

const port = chrome.runtime.connect({
    name: 'popup'
});

port.onMessage.addListener(message => {
    if (message.type == 'SW_TO_POPUP') {
        document.querySelector('#service-worker-to-popup').textContent = message.content;
    }
});

更新 service-worker.js 文件内容如下。

let popup_connect = null;

chrome.runtime.onConnect.addListener(port => {
    if (port.name == 'popup') {
        popup_connect = port;
        console.log('Connect!');
    }
    port.onDisconnect.addListener(() => popup_connect = null);
});

setInterval(() => {
    if (!popup_connect) {
        return console.log('Disconnect!');
    }
    popup_connect.postMessage({
        type: 'SW_TO_POPUP',
        content: new Date().toLocaleString()
    });
}, 1000);

更新 service-worker.js 文件内容如下。

chrome.runtime.onMessage.addListener((message, sender, sendResponse) => {
    switch (message.type) {
        case 'POPUP_TO_OPTIONS':
            chrome.runtime.sendMessage({
                type: 'POPUP_TO_OPTIONS',
                content: message.content
            });
            break;
        case 'OPTIONS_TO_POPUP':
            chrome.runtime.sendMessage({
                type: "OPTIONS_TO_POPUP",
                content: message.content
            });
            break;
        default:
            break;
    }
});

chrome.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
    if (changeInfo.status === 'complete' && tab.url) {
        chrome.tabs.sendMessage(tabId,
