## popup → service worker

更新 `popup.html` 文件内容如下。

```html
<!DOCTYPE html>
<html lang="zh-Hans">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Demo</title>
    <link rel="stylesheet" type="text/css" href="popup.css">
</head>
<body>
    <button id="popup-to-service-worker">Send</button>
    <h1 id="service-worker-response"></h1>
    <script type="text/javascript" src="popup.js"></script>
</body>
</html>
```

更新 `popup.css` 文件内容如下。

```css
body {
    text-align: center;
    width: 300px;
}

button {
    border: none;
    border-radius: 5px;
    color: #fff;
    cursor: pointer;
    font-size: 14px;
    font-weight: 500;
    margin: 10px;
    outline: 0;
    padding: 10px 20px;
    transition: all .2s ease;
    -webkit-tap-highlight-color: transparent;
}

button:hover {
    box-shadow: 0 2px 8px rgba(37, 99, 235, .2);
}

button:active {
    box-shadow: 0 1px 4px rgba(37, 99, 235, .15);
}

#popup-to-service-worker {
    background-color: #2563eb;
}

#popup-to-service-worker:hover {
    background-color: #1d4ed8;
}

#popup-to-service-worker:active {
    background-color: #1e40af;
}
```

更新 `popup.js` 文件内容如下。

```javascript
const popup_to_service_worker = document.querySelector('#popup-to-service-worker');
popup_to_service_worker.addEventListener('click', () => {
    chrome.runtime.sendMessage(
    {
        type: 'POPUP_TO_SW',
        content: 'Hello World!'
    },
    response => {
        const service_worker_response = document.querySelector('#service-worker-response');
        chrome.runtime.lastError ? service_worker_response.textContent = chrome.runtime.lastError : service_worker_response.textContent = response.content;
    }
  );
});
```

更新 `service-worker.js` 文件内容如下。

```javascript
chrome.runtime.onMessage.addListener((message, sender, sendResponse) => {
    if (message.type == 'POPUP_TO_SW') {
        console.log(message.content);
        sendResponse({
            content: message.content
        });
    }
});
```

## service worker → popup

更新 `popup.html` 文件内容如下。

```html
<!DOCTYPE html>
<html lang="zh-Hans">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Demo</title>
    <link rel="stylesheet" type="text/css" href="popup.css">
</head>
<body>
    <h1 id="service-worker-to-popup"></h1>
    <script type="text/javascript" src="popup.js"></script>
</body>
</html>
```

更新 `popup.css` 文件内容如下。

```css
body {
    text-align: center;
    width: 300px;
}
```

更新 `popup.js` 文件内容如下。

```javascript
const port = chrome.runtime.connect({
    name: 'popup'
});

port.onMessage.addListener(message => {
    if (message.type == 'SW_TO_POPUP') {
        document.querySelector('#service-worker-to-popup').textContent = message.content;
    }
});
```

更新 `service-worker.js` 文件内容如下。

```javascript
let popup_connect = null;

chrome.runtime.onConnect.addListener(port => {
    if (port.name == 'popup') {
        popup_connect = port;
        console.log('Connect!');
    }
    port.onDisconnect.addListener(() => popup_connect = null);
});

setInterval(() => {
    if (!popup_connect) {
        return console.log('Disconnect!');
    }
    popup_connect.postMessage({
        type: 'SW_TO_POPUP',
        content: new Date().toLocaleString()
    });
}, 1000);
```

```javascript
const options_to_popup = document.querySelector('#options-to-popup');
options_to_popup.addEventListener('click', () => {
    setInterval(() => {
        chrome.runtime.sendMessage({
            type: 'OPTIONS_TO_POPUP',
            content: new Date().toLocaleString()
        });
    }, 1000);
});

chrome.runtime.onMessage.addListener(message => {
    if (message.type == 'POPUP_TO_OPTIONS') {
        document.querySelector('#popup-to-options').textContent = message.content;
    }
});
```

更新 `service-worker.js` 文件内容如下。

```javascript
chrome.runtime.onMessage.addListener((message, sender, sendResponse) => {
    switch (message.type) {
        case 'POPUP_TO_OPTIONS':
            chrome.runtime.sendMessage({
                type: 'POPUP_TO_OPTIONS',
                content: message.content
            });
            break;
        case 'OPTIONS_TO_POPUP':
            chrome.runtime.sendMessage({
                type: "OPTIONS_TO_POPUP",
                content: message.content
            });
            break;
        default:
            break;
    }
});
```

## content ↔ service worker

更新 `content.js` 文件内容如下。

```javascript
chrome.runtime.sendMessage(
{
    type: 'CONTENT_TO_SW',
    content: 'Hello World!'
},
  response => console.log(response.content)
);

chrome.runtime.onMessage.addListener((message, sender, sendResponse) => {
  if (message.type == 'SW_TO_CONTENT') {
    console.log(message.content);
  }
  sendResponse({
    content: message.content
  });
});
```

更新 `service-worker.js` 文件内容如下。

```javascript
chrome.runtime.onMessage.addListener((message, sender, sendResponse) => {
    if (message.type == 'CONTENT_TO_SW') {
        console.log(message.content);
    }
    sendResponse({
        content: message.content
    });
});

chrome.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
    if (changeInfo.status === 'complete' && tab.url) {
        chrome.tabs.sendMessage(tabId,
        {
            type: 'SW_TO_CONTENT',
            content: 'Welcome！'
        },
        response => console.log(response.content)
        );
    }
});
```
