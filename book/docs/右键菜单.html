右键菜单

chrome.contextMenus 用于在浏览器右键菜单中添加自定义操作。更新 manifest.json 文件内容如下。

{
    "permissions": ["contextMenus"]
}

更新 service-worker.js 文件内容如下。

chrome.runtime.onInstalled.addListener(() => {
    chrome.contextMenus.create({
        id: 'menu',
        title: '右键菜单',
        contexts: ['all']
    });
    chrome.contextMenus.create({
        id: 'foo',
        title: '选项一',
        contexts: ['all'],
        parentId: 'menu'
    });
    chrome.contextMenus.create({
        id: 'bar',
        title: '选项二',
        contexts: ['all'],
        parentId: 'menu'
    });
    chrome.contextMenus.create({
        id: 'baz',
        title: '选项3',
        contexts: ['all'],
        parentId: 'bar'
    });
});

chrome.contextMenus.onClicked.addListener((info, tab) => {
    console.log(info, tab);
    switch (info.menuItemId) {
        case 'foo':
            console.log('foo');
            break;
        case 'baz':
            console.log('baz');
            break;
        default:
            break;
    }
});
